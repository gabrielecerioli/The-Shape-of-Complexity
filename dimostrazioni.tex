\subsection{Unicità della distribuzione stazionaria di un cammino stocastico}

\begin{boxl}{Teorema}
Sia $G$ un grafo non direzionato connesso e non bipartito e sia $\textbf{p}$ il vettore di probabilità tale che $\textbf{Wp}=\textbf{p}$\\
Allora $\mathbf{p}=\mathbf{\mathbf{\tilde{p}}}$
\end{boxl}

\begin{proof}
    
Sia $u$ un qualunque vertice che massimizzi
\begin{equation}
    \frac{\textbf{p}(u)}{k_u}
\end{equation}
Dall'ipotesi che $\textbf{Wp}=\textbf{p}$ si ottiene:
\begin{equation}
    \mathbf{p}(u)=\sum_{v:(v,u)\in E}^{}\frac{\mathbf{p}(v)}{k_v} \le \sum_{v:(v,u)\in E}^{}\frac{\mathbf{p}(u)}{k_u}=\mathbf{p}(u)
\end{equation}

Dunque deve valere che $\forall (v,u)$ connessi da un arco:
\begin{equation}
    \frac{\mathbf{p}(u)}{k_u}=\frac{\mathbf{p}(v)}{k_v}
    \label{eq_uniq_s_s_dis}
\end{equation}
Propaghiamo ora questa uguaglianza a tutto il grafo:\\

sia $S$ l'insieme dei vertici per cui vale \ref{eq_uniq_s_s_dis}. Supponiamo per assurdo che esista un vertice $w \notin S$.\\
Siccome il grafo è connesso, esiste un cammino da $u$ a $w$, dunque deve esserci un qualche arco $(y,z)$ tale per cui $y \in S$ e $z \notin S$.\\
Se ripetiamo la dimostrazione con il nodo $y$ al posto di $u$, otteniamo 
\begin{equation}
    \frac{\mathbf{p}(y)}{k_y}=\frac{\mathbf{p}(z)}{k_z}
\end{equation}
che è una contraddizione all'ipotesi che $y \in S$ e $z \notin S$.
\end{proof}

\subsection{Teorema di Kac e teoria ergodica}

\subsection{Disuguaglianza dell' autovalore massimo della matrice laplaciana}
\begin{boxl}{Teorema}
  Sia $G=(V,E)$ un grafo semplice e sia $k_{\text{max}}$ il grado massimo tra i suoi nodi. Allora vale la seguente disuguaglianza:
  \begin{equation}
    \lambda_n(G) \le 2 k_{\text{max}}
  \end{equation}
\end{boxl}
\begin{proof}
Cominciamo con l'applicare il teorema di Gerschgorin alla matrice $\mathbf{L}=\mathbf{D}-\mathbf{A}$:

    \begin{boxl}{Teorema di Gerschgorin}
    Per una matrice quadrata $M$, ogni autovalore $\lambda$ giace in almeno uno dei dischi nel piano complesso centrati nei valori della diagonale $M_{ii}$ con raggio $R_i=\sum_{j \not = i}^{}\left\lvert M_ij\right\rvert $, ovvero la somma dei valori assoluti degli altri elementi nella riga.
    \end{boxl}

Dalla definizione di $\mathbf{L}$ si ottiene:
\begin{equation}
    R_i=\sum_{j \not = i}^{}\left\lvert L_{ij}\right\rvert = k_i
\end{equation}

Per il teorema di Gerschgorin ogni autovalore deve soddisfare:
\begin{equation}
    \left\lvert \lambda -k_i\right\rvert \le k_i \Leftrightarrow  0 \le \lambda \le 2k_i
\end{equation}
Siccome questo vale per ogni autovalore, compreso il massimo $\lambda_n$ e dato che $k_i \le k_{\text{max}}$ $ \forall i$:
\begin{equation}
    \lambda_n \le 2 k_{\text{max}}
\end{equation}
\end{proof}
